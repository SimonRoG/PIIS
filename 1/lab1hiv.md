    має_аналіз(Людина, підвищений_рівень_лейкоцитів);
    має_аналіз(Людина, підвищений_рівень_лімфоцитів);
    має_аналіз(Людина, атипові_мононуклеарні_клітини).

    має_симптом(Людина, блювання).

має_симптоми(Людина, енцефаліт) :-
    має_симптом(Людина, нудота),
    має_симптом(Людина, блювання).

має_симптоми(Людина, езофагіт) :-
    має_симптом(Людина, біль_за_грудиною),
    має_симптом(Людина, порушення_ковтання).

% Правила для визначення рівня ймовірності ВІЛ-інфекції
% Дуже висока ймовірність ВІЛ
має_вірогідність_віл(Людина, дуже_висока) :-
    (
        має_симптоми(Людина, мононуклеоз),
        має_аналізи(Людина, мононуклеоз),
        має_симптоми(Людина, серозний_менінгіт),
        має_симптоми(Людина, енцефаліт),
        має_симптоми(Людина, езофагіт)
    );
    (
        кількість_симптомів_віл(Людина, K),
        K >= 15
    ).

% Висока ймовірність ВІЛ
має_вірогідність_віл(Людина, висока) :-
    (
        (
            має_симптоми(Людина, мононуклеоз),
            має_аналізи(Людина, мононуклеоз)
        );
        має_симптоми(Людина, серозний_менінгіт);
        має_симптоми(Людина, енцефаліт);
        має_симптоми(Людина, езофагіт)
    );
    (
        кількість_симптомів_віл(Людина, K),
        K < 15,
        К >= 10
    ).

% Середня ймовірність ВІЛ
має_вірогідність_віл(Людина, середня) :-
    (
        має_симптом(Людина, порушення_сну),
        має_симптом(Людина, тяжкість_у_підреберях),
        має_симптом(Людина, ниючі_болі_у_підреберях),
        має_симптом(Людина, понос),
        має_симптом(Людина, нудота),
        має_симптом(Людина, блювання)

        кількість_симптомів_віл(Людина, K),

        К >= 6

% Низька ймовірність ВІЛ
має_вірогідність_віл(Людина, низька) :-

        має_симптом(Людина, підвищена_температура),

% Підрахунок кількості симптомів ВІЛ у пацієнта

    findall(С, має_симптом(Людина, С), Список),
    length(Список, Кількість).
